cmake_minimum_required(VERSION 3.24)

project(Loki LANGUAGES CXX)

include(CTest)

option(LOKI_BUILD_TESTS "Build the Loki test suite" ON)
option(LOKI_ENABLE_COVERAGE "Enable coverage instrumentation" OFF)

set(CMAKE_CXX_STANDARD 23)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)

set(LOKI_ROOT_DIR "${CMAKE_CURRENT_SOURCE_DIR}")
set(LOKI_SOURCE_DIR "${LOKI_ROOT_DIR}/Loki")
set(LOKI_TEST_DIR "${LOKI_ROOT_DIR}/Loki.Tests")
set(LOKI_GENERATED_DIR "${CMAKE_CURRENT_BINARY_DIR}/generated")

file(MAKE_DIRECTORY "${LOKI_GENERATED_DIR}")

execute_process(
  COMMAND git rev-parse --short=8 HEAD
  WORKING_DIRECTORY "${LOKI_ROOT_DIR}"
  OUTPUT_VARIABLE LOKI_GIT_COMMIT
  OUTPUT_STRIP_TRAILING_WHITESPACE
  ERROR_QUIET
)
if(NOT LOKI_GIT_COMMIT)
  set(LOKI_GIT_COMMIT "unknown")
endif()

execute_process(
  COMMAND git rev-parse --abbrev-ref HEAD
  WORKING_DIRECTORY "${LOKI_ROOT_DIR}"
  OUTPUT_VARIABLE LOKI_GIT_BRANCH
  OUTPUT_STRIP_TRAILING_WHITESPACE
  ERROR_QUIET
)
if(NOT LOKI_GIT_BRANCH)
  set(LOKI_GIT_BRANCH "unknown")
endif()

file(WRITE "${LOKI_GENERATED_DIR}/__gitversion_autogen.hpp"
"// Auto-generated by CMake. Do not edit manually.\n"
"#pragma once\n"
"#define GIT_CUR_COMMIT \"${LOKI_GIT_COMMIT}\"\n"
"#define GIT_BRANCH \"${LOKI_GIT_BRANCH}\"\n")

set_property(GLOBAL PROPERTY USE_FOLDERS ON)

add_subdirectory(Loki)

if(LOKI_BUILD_TESTS)
  add_subdirectory(Loki.Tests)
endif()
